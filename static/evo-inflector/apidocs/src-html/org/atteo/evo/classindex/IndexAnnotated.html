014package org.atteo.evo.classindex;
015
016import java.lang.annotation.Documented;
017import java.lang.annotation.ElementType;
018import java.lang.annotation.Target;
019import java.util.ServiceLoader;
020
021import org.atteo.evo.classindex.processor.ClassIndexProcessor;
022
023/**
024 * Index all classes annotated by annotation annotated by this meta-annotation.
025 *
026 * <p>
027 * During compilation {@link ClassIndexProcessor} creates a resource file listing all classes
028 * annotated by annotation annotated by this meta-annotation.
029 * </p>
030 * <p>
031 * You can retrieve the list at runtime using {@link ClassIndex#getAnnotated(Class)}.
032 * If the classes also have a zero-argument constructor you can use {@link ServiceLoader} facility.
033 * </p>
034 */
035@Documented
036@Target(ElementType.ANNOTATION_TYPE)
037public @interface IndexAnnotated {
038    /**
039     * Specifies whether to store Javadoc for runtime retrieval.
040     *
041     * <p>
042     * You can retrieve the stored Javadoc summary using {@link ClassIndex#getSummary(Class)}.
043     * </p>
044     */
045    boolean storeJavadoc() default false;
046}
